****      Orion engine v1 MSH format datasheet             ***
**** Stalin Subway, Red Veil, Метро-2,Метро-2:Смерть вождя ***

1. Header, starts at 0x00000000

struct msh_head
{
	unsigned int type = 0x00000101;
	unsigned char name[16] = "Orion engine v1";
	unsigned int unk;
	unsigned int faces; // faces count
	unsigned int verts; // vertex count
};

2. Vertexes data, starts at 0x00000068, repeated msh_head.verts times

struct vector3f // size 12 = 3 * 4
{
	float x,y,z; 
};

3. Normales data, starts next, repeated msh_head.verts times

struct vector3f // size 12 = 3 * 4
{
	float x,y,z; 
};

4. Faces data, starts next, repeated msh_head.faces times

struct face  // size 6 = 3 * 2
{
	unsigned short f0,f1,f2; 
};

5. Texture UV coordinates data, starts next, repeated msh_head.verts times

struct vector2f // size 8 = 2 * 4
{
	float u,v; 
};

6. Texture DDS path next.

it is somewhere below, 
looks like this = "MODELS\PERSON\NPC_NEUT\MEDSESTRA\medsestra_tex.dds"
